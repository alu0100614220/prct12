def +(object)
  b = Racional.new(0,0)
  b.setden(mcm(@den,object.getden))
  b.setnum((b.getden/@den * @num)  + (b.getden/object.getden * object.getnum))
  b
end

def mcd(a,b)
  while b != 0
      a,b = b,a%b
  end
  a
end

def simplificar
    num = @num/mcd(@num,@den)
    den = @den/mcd(@num,@den)
    @num=num
    @den=den
end
